#include <iostream>
#include <vector>
#include <omp.h>
using namespace std;

/* Utility function to print array */
void printArray(vector<int> &a) {
    for (int x : a) cout << x << " ";
    cout << endl;
}

/* Sequential Bubble Sort */
void seqBubble(vector<int> &a) {
    int n = a.size();
    for (int i = 0; i < n - 1; i++)
        for (int j = 0; j < n - i - 1; j++)
            if (a[j] > a[j + 1])
                swap(a[j], a[j + 1]);
}

/* Parallel Bubble Sort (Odd-Even Sort) */
void parBubble(vector<int> &a) {
    int n = a.size();
    for (int i = 0; i < n; i++) {
        #pragma omp parallel for
        for (int j = i % 2; j < n - 1; j += 2) {
            if (a[j] > a[j + 1]) {
                swap(a[j], a[j + 1]);
            }
        }
    }
}

/* Merge Function */
void merge(vector<int> &a, int l, int m, int r) {
    vector<int> temp;
    int i = l, j = m + 1;

    while (i <= m && j <= r) {
        if (a[i] < a[j])
            temp.push_back(a[i++]);
        else
            temp.push_back(a[j++]);
    }

    while (i <= m) temp.push_back(a[i++]);
    while (j <= r) temp.push_back(a[j++]);

    for (int k = 0; k < temp.size(); k++)
        a[l + k] = temp[k];
}

/* Sequential Merge Sort */
void seqMerge(vector<int> &a, int l, int r) {
    if (l < r) {
        int m = (l + r) / 2;
        seqMerge(a, l, m);
        seqMerge(a, m + 1, r);
        merge(a, l, m, r);
    }
}

/* Parallel Merge Sort (Task-based) */
void parMerge(vector<int> &a, int l, int r) {
    if (l < r) {
        int m = (l + r) / 2;

        #pragma omp task shared(a)
        parMerge(a, l, m);

        #pragma omp task shared(a)
        parMerge(a, m + 1, r);

        #pragma omp taskwait
        merge(a, l, m, r);
    }
}

int main() {
    int n;
    cout << "Enter number of elements: ";
    cin >> n;

    vector<int> arr(n), temp;

    cout << "Enter elements:\n";
    for (int i = 0; i < n; i++)
        cin >> arr[i];

    double t1, t2;

    /* Sequential Bubble */
    temp = arr;
    t1 = omp_get_wtime();
    seqBubble(temp);
    t2 = omp_get_wtime();
    double seqBubbleTime = t2 - t1;

    cout << "\nSequential Bubble Sorted: ";
    printArray(temp);
    cout << "Time: " << seqBubbleTime << endl;

    /* Parallel Bubble */
    temp = arr;
    t1 = omp_get_wtime();
    parBubble(temp);
    t2 = omp_get_wtime();
    double parBubbleTime = t2 - t1;

    cout << "\nParallel Bubble Sorted: ";
    printArray(temp);
    cout << "Time: " << parBubbleTime << endl;

    cout << "Speedup (Bubble): " << seqBubbleTime / parBubbleTime << endl;

    /* Sequential Merge */
    temp = arr;
    t1 = omp_get_wtime();
    seqMerge(temp, 0, n - 1);
    t2 = omp_get_wtime();
    double seqMergeTime = t2 - t1;

    cout << "\nSequential Merge Sorted: ";
    printArray(temp);
    cout << "Time: " << seqMergeTime << endl;

    /* Parallel Merge */
    temp = arr;
    t1 = omp_get_wtime();

    #pragma omp parallel
    {
        #pragma omp single
        parMerge(temp, 0, n - 1);
    }

    t2 = omp_get_wtime();
    double parMergeTime = t2 - t1;

    cout << "\nParallel Merge Sorted: ";
    printArray(temp);
    cout << "Time: " << parMergeTime << endl;

    cout << "Speedup (Merge): " << seqMergeTime / parMergeTime << endl;

    return 0;
}
